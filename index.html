function makeMagicCircleTexture(size = 2048, hue = "170,235,255") { const canvas = document.createElement("canvas"); canvas.width = size; canvas.height = size; const ctx = canvas.getContext("2d"); const cx = size / 2; const cy = size / 2; ctx.clearRect(0, 0, size, size); // 背景グロー const bgGlow = ctx.createRadialGradient(cx, cy, 0, cx, cy, size * 0.49); bgGlow.addColorStop(0.0, `rgba(${hue},0.16)`); bgGlow.addColorStop(0.28, `rgba(${hue},0.08)`); bgGlow.addColorStop(0.55, `rgba(${hue},0.035)`); bgGlow.addColorStop(1.0, "rgba(0,0,0,0)"); ctx.fillStyle = bgGlow; ctx.beginPath(); ctx.arc(cx, cy, size * 0.49, 0, Math.PI * 2); ctx.fill(); ctx.strokeStyle = `rgba(${hue},0.96)`; ctx.lineWidth = size * 0.0022; ctx.shadowColor = `rgba(${hue},0.95)`; ctx.shadowBlur = size * 0.012; function ring(r, alpha = 1, width = 1) { ctx.save(); ctx.globalAlpha = alpha; ctx.lineWidth = size * 0.0022 * width; ctx.beginPath(); ctx.arc(cx, cy, r, 0, Math.PI * 2); ctx.stroke(); ctx.restore(); } function arcRing(r, count, arcRatio = 0.72, rotation = 0, alpha = 1, width = 1) { ctx.save(); ctx.globalAlpha = alpha; ctx.lineWidth = size * 0.0022 * width; for (let i = 0; i < count; i++) { const a0 = rotation + (Math.PI * 2 * i) / count; const a1 = a0 + (Math.PI * 2 / count) * arcRatio; ctx.beginPath(); ctx.arc(cx, cy, r, a0, a1); ctx.stroke(); } ctx.restore(); } function polygon(r, sides, rotation = 0, alpha = 1, width = 1) { ctx.save(); ctx.globalAlpha = alpha; ctx.lineWidth = size * 0.0022 * width; ctx.beginPath(); for (let i = 0; i < sides; i++) { const a = rotation + (Math.PI * 2 * i) / sides; const x = cx + Math.cos(a) * r; const y = cy + Math.sin(a) * r; if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y); } ctx.closePath(); ctx.stroke(); ctx.restore(); } function star(rOuter, rInner, points, rotation = 0, alpha = 1, width = 1) { ctx.save(); ctx.globalAlpha = alpha; ctx.lineWidth = size * 0.0022 * width; ctx.beginPath(); for (let i = 0; i < points * 2; i++) { const r = i % 2 === 0 ? rOuter : rInner; const a = rotation + (Math.PI * i) / points; const x = cx + Math.cos(a) * r; const y = cy + Math.sin(a) * r; if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y); } ctx.closePath(); ctx.stroke(); ctx.restore(); } function radialLines(r1, r2, count, rotation = 0, alpha = 1, width = 1) { ctx.save(); ctx.globalAlpha = alpha; ctx.lineWidth = size * 0.0022 * width; for (let i = 0; i < count; i++) { const a = rotation + (Math.PI * 2 * i) / count; const x1 = cx + Math.cos(a) * r1; const y1 = cy + Math.sin(a) * r1; const x2 = cx + Math.cos(a) * r2; const y2 = cy + Math.sin(a) * r2; ctx.beginPath(); ctx.moveTo(x1, y1); ctx.lineTo(x2, y2); ctx.stroke(); } ctx.restore(); } function tickRing(r, count, len, rotation = 0, alpha = 1, width = 1) { ctx.save(); ctx.globalAlpha = alpha; ctx.lineWidth = size * 0.002 * width; for (let i = 0; i < count; i++) { const a = rotation + (Math.PI * 2 * i) / count; const x1 = cx + Math.cos(a) * (r - len * 0.5); const y1 = cy + Math.sin(a) * (r - len * 0.5); const x2 = cx + Math.cos(a) * (r + len * 0.5); const y2 = cy + Math.sin(a) * (r + len * 0.5); ctx.beginPath(); ctx.moveTo(x1, y1); ctx.lineTo(x2, y2); ctx.stroke(); } ctx.restore(); } function runeDots(r, count, dotR, rotation = 0, alpha = 1) { ctx.save(); ctx.globalAlpha = alpha; ctx.fillStyle = `rgba(${hue},0.98)`; ctx.shadowColor = `rgba(${hue},1)`; ctx.shadowBlur = size * 0.012; for (let i = 0; i < count; i++) { const a = rotation + (Math.PI * 2 * i) / count; const x = cx + Math.cos(a) * r; const y = cy + Math.sin(a) * r; ctx.beginPath(); ctx.arc(x, y, dotR, 0, Math.PI * 2); ctx.fill(); } ctx.restore(); } function runeMarks(r, count, rotation = 0, alpha = 1, scale = 1) { ctx.save(); ctx.globalAlpha = alpha; ctx.lineWidth = size * 0.0018; for (let i = 0; i < count; i++) { const a = rotation + (Math.PI * 2 * i) / count; const x = cx + Math.cos(a) * r; const y = cy + Math.sin(a) * r; ctx.save(); ctx.translate(x, y); ctx.rotate(a + Math.PI / 2); ctx.scale(scale, scale); ctx.beginPath(); ctx.moveTo(0, -size * 0.010); ctx.lineTo(0, size * 0.010); ctx.moveTo(-size * 0.006, 0); ctx.lineTo(size * 0.006, 0); ctx.moveTo(0, -size * 0.010); ctx.lineTo(size * 0.006, -size * 0.016); ctx.moveTo(0, size * 0.010); ctx.lineTo(-size * 0.006, size * 0.016); ctx.stroke(); ctx.restore(); } ctx.restore(); } function orbitCircles(r, count, subR, rotation = 0, alpha = 1, width = 1) { ctx.save(); ctx.globalAlpha = alpha; ctx.lineWidth = size * 0.0018 * width; for (let i = 0; i < count; i++) { const a = rotation + (Math.PI * 2 * i) / count; const x = cx + Math.cos(a) * r; const y = cy + Math.sin(a) * r; ctx.beginPath(); ctx.arc(x, y, subR, 0, Math.PI * 2); ctx.stroke(); } ctx.restore(); } // 外周構成 ring(size * 0.460, 0.35, 0.9); ring(size * 0.442, 0.75, 1.5); arcRing(size * 0.425, 36, 0.58, 0, 0.70, 0.9); ring(size * 0.405, 0.65, 0.9); tickRing(size * 0.392, 72, size * 0.010, 0, 0.65, 0.8); // ルーン帯 ring(size * 0.360, 0.75, 1.2); runeMarks(size * 0.340, 24, Math.PI / 24, 0.92, 1.0); runeDots(size * 0.320, 24, size * 0.0038, 0, 0.85); ring(size * 0.300, 0.65, 0.9); // 放射構造 radialLines(size * 0.230, size * 0.395, 24, 0, 0.22, 0.8); radialLines(size * 0.140, size * 0.275, 12, Math.PI / 12, 0.45, 1.0); // 幾何学の核 polygon(size * 0.280, 6, -Math.PI / 2, 0.55, 1.0); star(size * 0.248, size * 0.155, 12, -Math.PI / 2, 0.80, 1.0); polygon(size * 0.188, 3, -Math.PI / 2, 0.72, 1.1); star(size * 0.145, size * 0.090, 6, Math.PI / 6, 0.68, 0.9); // 副円たち orbitCircles(size * 0.215, 6, size * 0.025, Math.PI / 6, 0.55, 0.9); orbitCircles(size * 0.115, 3, size * 0.018, -Math.PI / 2, 0.60, 0.9); // 内部リング ring(size * 0.132, 0.86, 1.0); arcRing(size * 0.112, 12, 0.50, Math.PI / 12, 0.68, 0.9); tickRing(size * 0.092, 24, size * 0.008, 0, 0.72, 0.8); ring(size * 0.070, 0.88, 1.0); // 中央コア const coreGlow = ctx.createRadialGradient(cx, cy, 0, cx, cy, size * 0.11); coreGlow.addColorStop(0.0, "rgba(255,255,255,1)"); coreGlow.addColorStop(0.18, `rgba(${hue},0.98)`); coreGlow.addColorStop(0.45, `rgba(${hue},0.34)`); coreGlow.addColorStop(1.0, `rgba(${hue},0)`); ctx.fillStyle = coreGlow; ctx.beginPath(); ctx.arc(cx, cy, size * 0.11, 0, Math.PI * 2); ctx.fill(); // 中央小紋章 ctx.save(); ctx.globalAlpha = 0.95; ctx.lineWidth = size * 0.002; ctx.beginPath(); for (let i = 0; i < 8; i++) { const a = -Math.PI / 2 + (Math.PI * 2 * i) / 8; const x = cx + Math.cos(a) * size * 0.048; const y = cy + Math.sin(a) * size * 0.048; if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y); } ctx.closePath(); ctx.stroke(); ctx.restore(); const tex = new THREE.CanvasTexture(canvas); tex.colorSpace = THREE.SRGBColorSpace; tex.needsUpdate = true; return tex; }
